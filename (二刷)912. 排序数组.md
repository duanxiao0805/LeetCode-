**还是用的快速排序**


```java
class Solution {
    public int[] sortArray(int[] arr) {
        quickSort(arr,0,arr.length-1);
        return arr;
    }
    //快速排序算法
    private void quickSort(int[] arr,int l,int r) {
        if(l>=r)
            return ;
        int x=arr[l];
        int i=l;
        int j=r;
        while(i<j) {
            while(j>i&&arr[j]>x){
                j--;
            }
            while(i<j&&arr[i]<=x) {
                i++;
            }
            swap(arr,i,j);
        }
        swap(arr,i,l);
        quickSort(arr,l,i-1);
        quickSort(arr,i+1,r);
    }
    private void swap(int [] arr,int i,int j) {
        //交换
        int temp=arr[i];
        arr[i]=arr[j];
        arr[j]=temp;
    }
}
```
