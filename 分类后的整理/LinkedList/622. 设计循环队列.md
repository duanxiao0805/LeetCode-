class MyCircularQueue {
        int maxSize;    //容量
        int front;      //指向队列的头部,头部数据
        int rear;       //指向队列尾部数据的后一个空位置
        int [] arr;
    
    
    //构造器构造
    public MyCircularQueue(int k) {
        maxSize=k+1;
        front=0;
        rear=0;
        arr =new int[maxSize];
    }
    
    public boolean enQueue(int value) {
        //判断是否满
        if(isFull()){
            return false;
        }
        arr[rear]=value;
        rear=(rear+1)%maxSize;
        
        return true;
    }
    
    public boolean deQueue() {
        //判断是否空
        if(isEmpty()){
            return false;
        }
        front=(front+1)%maxSize;
        return true;
    }
    
    public int Front() {
        //判断是否空
        if(isEmpty()){
            return -1;
        }
        int value;
        value=arr[front];
        return value;
    }
    
    public int Rear() {
        //判断是否空
        if(isEmpty()){
            return -1;
        }
        if(rear==0){
            return arr[arr.length-1];
        }
        return arr[rear];
    }
    //判断是否空
    public boolean isEmpty() {
        return front==rear;

    }
    //判断是否满
    public boolean isFull() {
        return (rear+1)%maxSize==front;
    }
}
