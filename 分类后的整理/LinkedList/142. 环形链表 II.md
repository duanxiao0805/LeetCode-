public class Solution {
    public ListNode detectCycle(ListNode head) {
        // 创建快慢指针
        ListNode fast=head,slow=head;
        // 进入循环
        while (true) {
            if (fast == null || fast.next == null) {
                return null;
            }
            fast = fast.next.next;
            slow = slow.next;
            if (fast == slow) {
                break;
            }
        }
        //更新快指针的节点位置
        fast=head;
        //第二次相遇,走一步
        while(fast!=slow){
            fast=fast.next;
            slow=slow.next;
        }
        return slow;
    }
}
