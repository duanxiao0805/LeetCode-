//这个题我想到了,数转换成链表的那个思路
class Solution {
    TreeNode head=new TreeNode(0);
    TreeNode real=head;
    public TreeNode increasingBST(TreeNode root) {
        if(root==null)
            return null;
        
        increasingBST(root.left);
        real.right=new TreeNode(root.val);
        real=real.right;
        increasingBST(root.right);

        return head.right;
    }
}
